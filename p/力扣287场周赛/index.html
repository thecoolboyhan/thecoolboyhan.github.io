class Solution {
    public int[] resultArray(int[] nums) {
        int n=nums.length;
        List<Integer> a=new ArrayList<>();
        List<Integer> b=new ArrayList<>();
        a.add(nums[0]);
        b.add(nums[1]);
        for(int i=2;i<n;i++){
            if(a.get(a.size()-1)>b.get(b.size()-1)){
                a.add(nums[i]);
            }else{
                b.add(nums[i]);
            }
        }
        a.addAll(b);
        for(int i=0;i<n;i++){
            nums[i]=a.get(i);
        }
        return nums;
    }
}

class Solution {

    public int countSubmatrices(int[][] grid, int k) {
        //æˆ‘çš„ä¸‘é™‹è§£æ³•
        int res=0;
        if(grid[0][0]>k) return res;
        int[] n1=new int[grid.length];
        n1[0]=grid[0][0];
        res++;
        int t= grid.length-1;
        for(int i=1;i<grid.length;i++){
            n1[i]=n1[i-1]+grid[i][0];
            res++;
            if(n1[i]>k){
                t=i;
                res--;
                break;
            }
        }
        for(int i=1;i<grid[0].length&&n1[0]<=k;i++){
            int s=0;
            for(int j=0;j<=t;j++){
                s+=grid[j][i];
                n1[j]+=s;
                if(n1[j]<=k) res++;
                else {
                    t=j;
                    break;
                }
            }
        }
        return res;
    }

}

class Solution {
    public int minimumOperationsToWriteY(int[][] grid) {
        // å’ŒèŒ¶ç¥æƒ³æ³•ä¸€æ ·ï¼Œä½†å°±æ˜¯èœæ²¡åšå‡ºæ¥
        int[] t1=new int[3];
        int[] t2=new int[3];
        for(int i=0;i<grid.length/2;i++){
            for(int j=0;j<grid.length;j++){
                if(i==j||i==grid.length-1-j){
                    t1[grid[i][j]]++;
                } else{
                    t2[grid[i][j]]++;
                }
            }
        }
        for(int i=grid.length/2;i<grid.length;i++){
            for(int j=0;j<grid.length;j++){
                if(j==grid.length/2) t1[grid[i][j]]++;
                else t2[grid[i][j]]++;
            }
        }
        int res=0;
        for(int i=0;i<3;i++){
            for(int j=0;j<3;j++){
                if(i!=j){
                    res=Math.max(res,t1[i]+t2[j]);
                }
            }
        }
        return grid.length*grid[0].length-res;
    }
}

class Fenwick{
    private final int[] tree;

    public Fenwick(int n){
        tree=new int[n];
    }

    // æŠŠä¸‹æ ‡ä¸ºiçš„å…ƒç´ å¢åŠ v
    public void add(int i,int v){
        while(i<tree.length){
            tree[i]+=v;
            // æ±‚lowbit
            i+=i&-i;
        }
    }

//  è¿”å›ä¸‹æ ‡åœ¨ã€1ï¼Œiã€‘çš„å…ƒç´ ä¹‹å’Œ
    public int pre(int i){
        int res=0;
        while(i>0){
            res+=tree[i];
//          è¿™ä¸ªæœ‰ç‚¹å¤æ‚
            i&=i-1;
        }
        return res;
    }
}

class Solution {
    public int[] resultArray(int[] nums) {
        // æŠ„äº†èŒ¶ç¥çš„æ ‘çŠ¶æ•°ç»„ï¼Œæš‚æ—¶å¬æ‡‚äº†ï¼Œä½†è‡ªå·±æ“ä½œè¿˜æ˜¯æä¸æ¥
        int[] sorted=nums.clone();
        Arrays.sort(sorted);//æ²¡æœ‰å»é‡çš„æ’åº
        int n=nums.length;
        List<Integer> a=new ArrayList<>(n);
        List<Integer> b=new ArrayList<>();
        a.add(nums[0]);
        b.add(nums[1]);
        Fenwick t=new Fenwick(n+1);
        t.add(n-Arrays.binarySearch(sorted,nums[0]),1);
        t.add(n-Arrays.binarySearch(sorted,nums[1]),-1);

        for(int i=2;i<nums.length;i++){
            int x=nums[i];
            int v=n-Arrays.binarySearch(sorted,x);
            int d=t.pre(v-1);  //è½¬æ¢æˆ<vçš„å…ƒç´ ä¸ªæ•°ä¹‹å·®
            if(d>0||d==0&&a.size()<=b.size()){
                a.add(x);
                t.add(v,1);
            }else{
                b.add(x);
                t.add(v,-1);
            }
        }
        a.addAll(b);
        for(int i=0;i<n;i++){
            nums[i]=a.get(i);
        }
        return nums;
    }
}
