static int[] prefix_function(String s){
    int n=s.length();
    int[] pi=new int[n];
    for(int i=1;i<n;i++){
        for(int j=i;j>=0;j--){
            if(s.substring(0,j).equals(s.substring(i-j+1,i+1))){
                pi[i]=j;
                break;
            }
        }
    }
    return pi;
}

static int[] prefix_function(String s){
    int n=s.length();
    int[] pi=new int[n];
    for(int i=1;i<n;i++){
        int j=pi[i-1];
        while(j>0&&s.charAt(i)!=s.charAt(j)){
            j=pi[j-1];
        }
        if(s.charAt(i)==s.charAt(j)){
            j++;
        }
        pi[i]=j;
    }
    return pi;
}
